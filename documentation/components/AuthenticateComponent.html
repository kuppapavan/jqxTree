import { Component,  OnInit,  Input,
         EventEmitter,  Output,  HostListener,
         OnChanges, SimpleChanges, ViewChild, ElementRef, QueryList } from '@angular/core';
import { Router, ActivatedRoute, ParamMap } from '@angular/router';
import { HttpClientService } from '../services/http-apis.service';
import { URLS } from '../enums/service-urls.enum';

@Component({
  selector: 'app-authenticate',
  templateUrl: './authenticate.component.html',
  styleUrls: ['./authenticate.component.css'],
  providers : [HttpClientService]
})
export class AuthenticateComponent implements OnInit {

  userName: string;
  password: string;
  loginStatus = false;
  postData : any;
  constructor(private httpClient: HttpClientService,
              private route: ActivatedRoute,
              private router: Router) {    
   }
  ngOnInit() {
    
  }

  doLogin() {
    const json = [{
        'username': this.userName,
        'password': this.password
      } ];
       this.httpClient.postRequest(URLS.LOGINURL ,json).subscribe(data => {
        this.postData = data
        if(this.postData.status=="Success"){
          this.router.navigate(['/home']); 
        }             
        }); 
    }
}
