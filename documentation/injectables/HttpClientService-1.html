@Injectable({
  providedIn: 'root'
})
export class HttpClientService {
  
   contextpath : string;
   httpdata ;
   httpOptions = {
    headers: new HttpHeaders({
      'Content-Type':  'application/json'      
    })
  };  
  endpoint = 'http://10.20.201.169:8082/iBemsServer/';
  constructor(private sharedService: ShareCommonObjectsService,
    private http: HttpClient) {

  getRequest(logInurl): Observable<any> {
    logInurl = this.endpoint + logInurl;
     return this.http.get(logInurl).pipe(map(this.extractData));
  }

  private extractData(res: Response) {
  let body = res;
  return body ;
  }

  public postRequest (baseUrl : string, jsonObj){
    baseUrl = this.endpoint + baseUrl;
    return this.http.post(baseUrl, jsonObj, this.httpOptions);
  }
}
